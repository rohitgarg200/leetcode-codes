class Solution {
public:
    int findJudge(int N, vector<vector<int>>& trust) {
            vector<int>a(N,0);
            for(int i=0;i<trust.size();i++){
                    a[trust[i][1]-1]++;
                    a[trust[i][0]-1]--;
            }
            for(int i=0;i<N;i++){
             if(a[i]==N-1){
                     return i+1;
             }       
            }
        return -1;
    }
};
