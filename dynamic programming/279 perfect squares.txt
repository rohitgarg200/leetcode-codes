class Solution {
public:
    int numSquares(int n) {
        vector<int>a(n+1,1000);
        a[1]=1;
        a[0]=0;
        for(int i=1;i<=n;i++){
            for(int j=1;j<=sqrt(i);j++){
                a[i]=min(a[i],a[i-pow(j,2)]+1);
            }
        }
        return a[n];
        
    }
};
